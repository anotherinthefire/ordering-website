if ($result->num_rows > 0) {
    $row = $result->fetch_assoc();

    // Display product information
    echo '<div class="box">';
    echo '<div class="product_image"><img src="' . $row["prod_img"] . '" alt=""></div>';
    echo '<div class="content">';
    echo '<h3>' . $row["prod_name"] . '</h3>';
    echo '<h2 class="price">₱ ' . number_format($row["prod_price"], 2) . '</h2>';
    echo nl2br('<p>' . $row["prod_desc"] . '</p>');
    echo '</div>';
    echo '</div>';

    // Retrieve available sizes and colors for the product from the stock table
    $sql = "SELECT size.size, color.color
            FROM stock
            INNER JOIN size ON stock.size_id = size.size_id
            INNER JOIN color ON stock.color_id = color.color_id
            WHERE stock.prod_id = $prod_id";
    $result = $conn->query($sql);

    if ($result->num_rows > 0) {
        echo '<h4>Available Sizes:</h4>';
        echo '<ul>';
        while ($row = $result->fetch_assoc()) {
            echo '<li>' . $row["size"] . '</li>';
        }
        echo '</ul>';

        $result->data_seek(0); // reset the result set to the first row

        echo '<h4>Available Colors:</h4>';
        echo '<ul>';
        while ($row = $result->fetch_assoc()) {
            echo '<li>' . $row["color"] . '</li>';
        }
        echo '</ul>';
    }
    else {
        echo '<p>No sizes and colors available for this product.</p>';
    }
} 
else {
    echo '<p>Product not found.</p>';
}

$conn->close();
